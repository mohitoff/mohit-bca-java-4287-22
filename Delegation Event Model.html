9. Java Program to Create and Switch between Frames using Buttons

import javax.swing.*;
import java.awt.event.*;

public class SwitchFrames {
    public static void main(String[] args) {
        JFrame frame1 = new JFrame("Frame 1");
        JButton button = new JButton("Go to Frame 2");
        frame1.add(button);
        frame1.setSize(300, 200);
        frame1.setVisible(true);
        frame1.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        
        JFrame frame2 = new JFrame("Frame 2");
        JButton backButton = new JButton("Go back to Frame 1");
        frame2.add(backButton);
        frame2.setSize(300, 200);
        
        button.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                frame1.setVisible(false);
                frame2.setVisible(true);
            }
        });

        backButton.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                frame2.setVisible(false);
                frame1.setVisible(true);
            }
        });
    }
}

13. Java Program to Close the Frame using WindowAdapter Class

import javax.swing.*;
import java.awt.event.*;

public class CloseFrame {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Close Frame Example");
        frame.setSize(300, 200);
        frame.setVisible(true);
        frame.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
        frame.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                int response = JOptionPane.showConfirmDialog(frame, "Do you really want to close?");
                if (response == JOptionPane.YES_OPTION) {
                    System.exit(0);
                }
            }
        });
    }
}

18. Java Program to Display Several Dots on the Screen Continuously

19. Java Program to Validate the TextField for only entering numbers using KeyListener

import javax.swing.*;
import java.awt.event.*;

public class NumberValidation {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Number Validation");
        JTextField textField = new JTextField(10);
        textField.addKeyListener(new KeyAdapter() {
            public void keyTyped(KeyEvent e) {
                if (!Character.isDigit(e.getKeyChar())) {
                    e.consume(); // Ignore non-digit input
                }
            }
        });
        frame.add(textField);
        frame.setSize(300, 200);
        frame.setVisible(true);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }
}
