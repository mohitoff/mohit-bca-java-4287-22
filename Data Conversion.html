Data Conversion Programs

1. How to convert String to int

String str = "123";
int num = Integer.parseInt(str); // Converts String to int

2. How to convert int to String

int num = 123;
String str = Integer.toString(num); // Converts int to String

3. How to convert String to long

String str = "123456789";
long num = Long.parseLong(str); // Converts String to long

4. How to convert long to String

long num = 123456789L;
String str = Long.toString(num); // Converts long to String

5. How to convert String to float

String str = "123.45";
float num = Float.parseFloat(str); // Converts String to float

6. How to convert float to String

float num = 123.45f;
String str = Float.toString(num); // Converts float to String

7. How to convert String to double

String str = "123.45";
double num = Double.parseDouble(str); // Converts String to double

8. How to convert double to String

double num = 123.45;
String str = Double.toString(num); // Converts double to String

9. How to convert String to Date

import java.text.SimpleDateFormat;
import java.util.Date;

String str = "2024-10-31";
SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
Date date = sdf.parse(str); // Converts String to Date

10. How to convert Date to String

import java.text.SimpleDateFormat;
import java.util.Date;

Date date = new Date();
SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
String str = sdf.format(date); // Converts Date to String

11. How to convert String to char

String str = "Hello";
char ch = str.charAt(0); // Converts String to char (first character)

12. How to convert char to String

char ch = 'H';
String str = Character.toString(ch); // Converts char to String

13. How to convert String to Object

String str = "Hello";
Object obj = str; // Implicitly converts String to Object

14. How to convert Object to String

Object obj = "Hello";
String str = (String) obj; // Explicitly converts Object to String

15. How to convert int to long

int num = 123;
long longNum = (long) num; // Converts int to long

16. How to convert long to int

long longNum = 123456789L;
int num = (int) longNum; // Converts long to int

17. How to convert int to double

int num = 123;
double doubleNum = (double) num; // Converts int to double

18. How to convert double to int

double doubleNum = 123.45;
int num = (int) doubleNum; // Converts double to int (truncates decimal)

19. How to convert char to int

char ch = 'A';
int num = (int) ch; // Converts char to int (ASCII value)

20. How to convert int to char

int num = 65;
char ch = (char) num; // Converts int to char (ASCII value)

21. How to convert String to boolean

String str = "true";
boolean bool = Boolean.parseBoolean(str); // Converts String to boolean

22. How to convert boolean to String

boolean bool = true;
String str = Boolean.toString(bool); // Converts boolean to String

23. How to convert Date to Timestamp

import java.sql.Timestamp;
import java.util.Date;

Date date = new Date();
Timestamp timestamp = new Timestamp(date.getTime()); // Converts Date to Timestamp

24. How to convert Timestamp to Date

import java.sql.Timestamp;
import java.util.Date;

Timestamp timestamp = new Timestamp(System.currentTimeMillis());
Date date = new Date(timestamp.getTime()); // Converts Timestamp to Date

25. How to convert Binary to Decimal

String binaryStr = "1010";
int decimal = Integer.parseInt(binaryStr, 2); // Converts Binary to Decimal

26. How to convert Decimal to Binary

int decimal = 10;
String binaryStr = Integer.toBinaryString(decimal); // Converts Decimal to Binary

27. How to convert Hex to Decimal

String hexStr = "A";
int decimal = Integer.parseInt(hexStr, 16); // Converts Hex to Decimal

28. How to convert Decimal to Hex

int decimal = 10;
String hexStr = Integer.toHexString(decimal); // Converts Decimal to Hex
