Graphical User Interface (AWT) Programs

1. Simple AWT Application with Button (FlowLayout)

import java.awt.*;
import java.awt.event.*;

public class ButtonFlowLayout {
    public static void main(String[] args) {
        Frame frame = new Frame("Button with FlowLayout");
        frame.setLayout(new FlowLayout());
        
        Button button = new Button("Click Me");
        frame.add(button);
        
        frame.setSize(300, 200);
        frame.setVisible(true);
        
        frame.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent we) {
                System.exit(0);
            }
        });
    }
}

This AWT application creates a button using FlowLayout.

3. Multiple Buttons with BorderLayout

import java.awt.*;
import java.awt.event.*;

public class BorderLayoutButtons {
    public static void main(String[] args) {
        Frame frame = new Frame("BorderLayout Example");
        frame.setLayout(new BorderLayout());
        
        frame.add(new Button("North"), BorderLayout.NORTH);
        frame.add(new Button("South"), BorderLayout.SOUTH);
        frame.add(new Button("East"), BorderLayout.EAST);
        frame.add(new Button("West"), BorderLayout.WEST);
        frame.add(new Button("Center"), BorderLayout.CENTER);
        
        frame.setSize(400, 300);
        frame.setVisible(true);
        
        frame.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent we) {
                System.exit(0);
            }
        });
    }
}

This application displays multiple buttons using BorderLayout.

18. AWT Calculator Application with GridLayout

import java.awt.*;
import java.awt.event.*;

public class Calculator {
    public static void main(String[] args) {
        Frame frame = new Frame("Calculator");
        frame.setLayout(new GridLayout(4, 4));
        
        TextField display = new TextField();
        frame.add(display);
        
        // Adding buttons for calculator
        for (int i = 0; i < 10; i++) {
            frame.add(new Button(String.valueOf(i)));
        }
        
        frame.add(new Button("+"));
        frame.add(new Button("-"));
        frame.add(new Button("*"));
        frame.add(new Button("/"));
        
        frame.setSize(300, 300);
        frame.setVisible(true);
        
        frame.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent we) {
                System.exit(0);
            }
        });
    }
}

This application designs a simple calculator layout using GridLayout.
