1. Accessing java.util.Date Class via Import

import java.util.Date;

public class CurrentDateImport {
    public static void main(String[] args) {
        Date currentDate = new Date();
        System.out.println("Current Date: " + currentDate);
    }
}

2. Accessing java.util.Date Class via Qualified Way

    public static void main(String[] args) {
        java.util.Date currentDate = new java.util.Date();

    }
}

5. Creating a Package with Multiple Default Classes

package mypackage;

class ClassOne {
    void display() {
        System.out.println("Class One");
    }
}

class ClassTwo {
    void display() {
        System.out.println("Class Two");
    }
}

class ClassThree {
    void display() {
        System.out.println("Class Three");
    }
}

class ClassFour {
    void display() {
        System.out.println("Class Four");
    }
}

public class PackageDemo {
    public static void main(String[] args) {
        ClassOne c1 = new ClassOne();
        ClassTwo c2 = new ClassTwo();
        ClassThree c3 = new ClassThree();
        ClassFour c4 = new ClassFour();

        c1.display();
        c2.display();
        c3.display();
        c4.display();
    }
}

6. Creating a Package with 4 Public Classes

package mypackage;

public class ClassA {
    public void display() {
        System.out.println("Class A");
    }
}

public class ClassB {
    public void display() {
        System.out.println("Class B");
    }
}

public class ClassC {
    public void display() {
        System.out.println("Class C");
    }
}

public class ClassD {
    public void display() {
        System.out.println("Class D");
    }
}

// Save this as another file (e.g., MainClass.java)
import mypackage.*;

public class MainClass {
    public static void main(String[] args) {
        ClassA a = new ClassA();
        ClassB b = new ClassB();
        ClassC c = new ClassC();
        ClassD d = new ClassD();

        a.display();
        b.display();
        c.display();
        d.display();
    }
}

7. Employee Payroll Processing Using Packages

package employee;

public class Emp {
    String name;
    String empid;
    String category;
    double bpay, hra, da, npay, pf, grosspay, incometax, allowance;

    public void calculatePay() {
        // Sample calculations
        grosspay = bpay + hra + da;
        npay = grosspay - (pf + incometax);
    }
}

// Save this as another file (e.g., Emppay.java)
import employee.Emp;

public class Emppay {
    public static void main(String[] args) {
        Emp e = new Emp();
        e.name = "John Doe";
        e.empid = "E001";
        e.bpay = 30000;
        e.hra = 5000;
        e.da = 2000;
        e.pf = 2000;
        e.incometax = 500;
        e.calculatePay();

        System.out.println("Employee Name: " + e.name);
        System.out.println("Net Pay: " + e.npay);
        System.out.println("Gross Pay: " + e.grosspay);
    }
}

8. Package with Public Class and Protected Members

package mypackage;

public class PublicClass {
    protected String message = "Protected Message";

    public void showMessage() {
        System.out.println(message);
    }
}

// Save this as another file (e.g., AccessProtected.java)
import mypackage.PublicClass;

public class AccessProtected extends PublicClass {
    public static void main(String[] args) {
        AccessProtected obj = new AccessProtected();
        obj.showMessage();
    }
}
