1. Find the Smallest Number Among Three Numbers

public class SmallestNumber {
    public static int findSmallest(int a, int b, int c) {
        return Math.min(a, Math.min(b, c));
    }

    public static void main(String[] args) {
        System.out.println("Smallest number: " + findSmallest(10, 20, 5));
    }
}

This method takes three integers as parameters and returns the smallest among them using the Math.min method.

3. Display the Middle Character of a String

public class MiddleCharacter {
    public static String getMiddleCharacter(String str) {
        int length = str.length();
        if (length % 2 == 0) {
            return String.valueOf(str.charAt(length / 2 - 1));
        } else {
            return str.substring(length / 2 - 1, length / 2 + 1);
        }
    }

    public static void main(String[] args) {
        System.out.println("Middle character(s): " + getMiddleCharacter("example"));
        System.out.println("Middle character(s): " + getMiddleCharacter("test"));
    }
}

This method checks the length of the string and returns either one or two middle characters based on its length.

4. Count All the Words in a String

public class WordCounter {
    public static int countWords(String str) {
        String[] words = str.trim().split("\\s+");
        return words.length;
    }

    public static void main(String[] args) {
        System.out.println("Word count: " + countWords("Hello world, this is a test."));
    }
}

This method counts the number of words in a given string by splitting the string at whitespace.

5. Compute the Future Investment Value

public class InvestmentCalculator {
    public static double computeFutureValue(double investment, double interestRate, int years) {
        return investment * Math.pow((1 + interestRate / 100), years);
    }

    public static void main(String[] args) {
        System.out.println("Future investment value: " + computeFutureValue(1000, 5, 10));
    }
}

This method computes the future investment value based on the initial investment, interest rate, and number of years.

6. Print Characters Between Two Characters

public class CharacterRange {
    public static void printCharacters(char start, char end) {
        for (char ch = start; ch <= end; ch++) {
            System.out.print(ch + " ");
        }
    }

    public static void main(String[] args) {

    }
}

This method prints all characters between two specified characters (inclusive).

7. Check Whether a String is a Valid Password

public class PasswordValidator {
    public static boolean isValidPassword(String password) {
        return password.length() >= 8 && password.matches(".*[!@#$%^&*].*");
    }

    public static void main(String[] args) {
        System.out.println("Is valid password: " + isValidPassword("Password123!"));
    }
}

This method checks if the password is at least 8 characters long and contains a special character.

9. Find All Twin Prime Numbers Less Than 100

public class TwinPrimes {
    public static void findTwinPrimes() {
        for (int i = 3; i < 100; i++) {
            if (isPrime(i) && isPrime(i + 2)) {
                System.out.println("(" + i + ", " + (i + 2) + ")");
            }
        }
    }

    public static boolean isPrime(int num) {
        if (num <= 1) return false;
        for (int i = 2; i <= Math.sqrt(num); i++) {
            if (num % i == 0) return false;
        }
        return true;
    }

    public static void main(String[] args) {
        findTwinPrimes();
    }
}

This method finds and prints all twin prime numbers less than 100.

11. Check Whether Three Integers are Consecutive

public class ConsecutiveChecker {
    public static boolean areConsecutive(int a, int b, int c) {
        return (Math.abs(a - b) == 1 && Math.abs(b - c) == 1) || (Math.abs(a - c) == 1 && Math.abs(c - b) == 1);
    }

    public static void main(String[] args) {
        System.out.println("Are consecutive: " + areConsecutive(4, 5, 6));
    }
}

This method checks if three integers are consecutive and returns true or false.

13. Check Whether All Characters in a String are Vowels

public class VowelChecker {
    public static boolean areAllVowels(String str) {
        return str.chars().allMatch(ch -> "aeiouAEIOU".indexOf(ch) != -1);
    }

    public static void main(String[] args) {
        System.out.println("All characters are vowels: " + areAllVowels("aeiou"));
    }
}

This method checks if all characters in a string are vowels and returns true or false.
